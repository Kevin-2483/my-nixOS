# proxy-groups

```javascript
// https://mihomo.party/docs/guide/override/javascript
// 操作函数：动态修改代理组
function main(config) {
  let allProxies = config.proxies.map(proxy => proxy.name); // 获取所有代理的名称
  let flagGroups = {}; // 用来存储每个旗帜符号对应的代理组
  
  // 遍历所有代理节点名称，提取其中的旗帜符号
  allProxies.forEach(proxy => {
    // 匹配代理节点中的旗帜符号（假设旗帜符号出现在代理名称的前面）
    const flag = proxy.match(/[\uD83C-\uDBFF\uDC00-\uDFFF]+/g)?.[0]; // 使用正则提取旗帜符号
    if (flag) {
      // 如果找到旗帜符号，将节点添加到对应的旗帜组
      if (!flagGroups[flag]) {
        flagGroups[flag] = [];
      }
      flagGroups[flag].push(proxy);
    }
  });

  // 创建代理组
  const proxyGroups = [];
  
  // 根据每个旗帜符号生成代理组
  Object.keys(flagGroups).forEach(flag => {
    const nameAuto = flag + "自动选择";
    const nameFall = flag + "故障转移";
    const nameSelect = flag + "固定";
    proxyGroups.push({
      name: nameSelect, // 组名就是旗帜符号
      type: 'select', // 代理组类型，'select' 表示自动选择
      proxies: flagGroups[flag], // 将所有相同旗帜的代理节点添加到该组
    },
    {
      name: nameAuto, // 组名就是旗帜符号
      type: 'url-test',
      proxies: flagGroups[flag], // 将所有相同旗帜的代理节点添加到该组
      url: 'http://www.gstatic.com/generate_204',
      interval: 300,
    },
    {
      name: nameFall, // 组名就是旗帜符号
      type: 'fallback',
      proxies: flagGroups[flag], // 将所有相同旗帜的代理节点添加到该组
      url: 'http://www.gstatic.com/generate_204',
      interval: 300,
    });
  });

  // 自动生成 "Auto-Select" 代理组
  proxyGroups.push({
    name: '全部 自动选择',
    type: 'url-test',
    proxies: allProxies,
    url: 'http://www.gstatic.com/generate_204',
    interval: 300,
  });

  // 自动生成 "Failover" 代理组
  proxyGroups.push({
    name: '全部 故障转移',
    type: 'fallback',
    proxies: allProxies,
    url: 'http://www.gstatic.com/generate_204',
    interval: 300,
  });

  // 生成 "Default" 组
  // Default 组将包含 allProxies 和其他所有组，但不包括 Default 组本身
  const defaultProxies = proxyGroups.filter(group => group.name !== 'Default').map(group => group.name);
  
  // 最后添加 Default 组
  proxyGroups.push({
    name: 'Default',
    type: 'select',
    proxies: defaultProxies, // 将所有代理（包括所有旗帜组和 allProxies）加入 Default 组
  });

  // 返回更新后的代理组列表
  config['proxy-groups'] = config['proxy-groups'] || [];
  proxyGroups.forEach(group => {
    config['proxy-groups'].unshift(group); // 每个代理组单独插入
  });

  return config;
}
```
